	export default defineComponent({

			Home : defineAsyncComponent(()=> import('../views/Home.vue') ),
			Blog : defineAsyncComponent(()=> import('../views/Blog.vue') ),
		},
		computed:{
			fullPath:function(){
				return this.$route.path;
			}
		}
	})
</script>
